Grid多选列:checkBox

默认为false,不显示 | true 显示
关联属性(默认值):
checkBoxWidth : '20px' 多选列宽度
checkBoxTitle : '<input type="checkbox">' 多选列内容(一般不需要修改这个)
checkBoxFit : false 多选列是否可改变大小
注:
checkBox 为true的时候 关联属性才生效，而且columns中的 ck 会成为系统字段

Script: 
var data = [
	{id:1,title:'test1',content:'内容1',path:'base1.html'},
	{id:2,title:'test2',content:'内容2',path:'base2.html'},
	{id:3,title:'test3',content:'内容3',path:'base3.html'},
	{id:4,title:'test4',content:'内容4',path:'base4.html'},
	{id:5,title:'test5',content:'内容5',path:'base5.html'},
	{id:6,title:'test6',content:'内容6',path:'base6.html'},
	{id:7,title:'test7',content:'内容7',path:'base7.html'}
];

//Cfg建议写道函数里
function getCfg(){
	var columns = [
		{field:'id',title:'ID',width:30,fitColumn:false},
		{field:'title',title:'标题',width:60},
		{field:'content',title:'内容',width:200},
		{field:'path',title:'文件名',width:100}
	];
	var options = {
		title : 'gird checkBox',
		data : data,
		columns : columns,
		checkBox :　true,
		checkBoxWidth : '25px',
		checkBoxFit : true
	}
	return options;
}
function getCfg2(){
	var columns = [
		{field:'id',title:'ID',width:30,fitColumn:false},
		{field:'title',title:'标题',width:60},
		{field:'content',title:'内容',width:200},
		{field:'path',title:'文件名',width:100},
		{field:'ck'}
	];
	var options = {
		title : 'gird checkBox自定义显示位置',
		data : data,
		columns : columns,
		checkBox :　true,
		checkBoxWidth : '25px',
		checkBoxFit : false
	}
	return options;
}
var grid1,grid2;
function create(){
	var opt1 = getCfg();
	grid1 = $("#demo1").extgrid(opt1);
	var opt2 = getCfg2();
	grid2 = $("#demo2").extgrid(opt2);
}
$(function(){
	create();
});
