Grid编辑列:editColumn

默认为false,不显示 | [object1,object2,object3,...]
object=>{text : '',cls : '',callBack : $.noop,disabled:false}(设置可参考toolBars)
关联属性(默认值):
editCellW : 60 每个按钮宽度
editColumnTitle : '操作' 编辑列标题
editColumnFit : true 编辑列是否可改变大小
注:
callBack(t(按钮本身),rowId(行id),field(字段包含的信息),rowData(所有数据) )
editColumn 为true的时候 关联属性才生效，而且columns中的 ed 会成为系统字段

Script:

//Cfg建议写道函数里
function getCfg(){
	var columns = [
		{field:'id',title:'ID',width:30,fitColumn:false},
		{field:'title',title:'标题',width:60},
		{field:'content',title:'内容',width:200},
		{field:'path',title:'文件名',width:100}
	];
	var options = {
		title : 'gird editColumn',
		data : data,
		columns : columns,
		editColumn :　[ {text:'编辑',callBack:function(t,rowId,field,rowData){alert(rowId);}},"|",{text:"删除"},"|","我不是按钮" ]
	}
	return options;
}
function getCfg2(){
	var columns = [
		{field:'ed'},
		{field:'id',title:'ID',width:30,fitColumn:false},
		{field:'title',title:'标题',width:60},
		{field:'content',title:'内容',width:200},
		{field:'path',title:'文件名',width:100}
		
	];
	var options = {
		title : 'gird editColumn自定义显示位置',
		data : data,
		columns : columns,
		editColumn :　[ {text:'编辑'},"|",{text:"删除"},"|","我不是按钮" ]
	}
	return options;
}
var grid1,grid2;
function create(){
	var opt1 = getCfg();
	grid1 = $("#demo1").extgrid(opt1);
	var opt2 = getCfg2();
	grid2 = $("#demo2").extgrid(opt2);
}

	create();
