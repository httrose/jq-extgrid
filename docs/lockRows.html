Grid 行锁:lockRows

默认为 [] 行ID
注:
可调用 lockRow(rowId) 实现行锁
行锁只锁定行 不锁定行数据

Script: 
var data = [
	{id:1,title:'test1',content:'内容1234566789',path:'base1.html'},
	{id:2,title:'test2',content:'内容2tyuiopf',path:'base2.html'},
	{id:3,title:'test3',content:'我被锁定内容3zxcvbnm',path:'base3.html'},
	{id:4,title:'test4',content:'我被锁定内容qwertyui4',path:'base4.html'},
	{id:5,title:'test2',content:'内容2tyuiopf',path:'base2.html'},
	{id:6,title:'test2',content:'内容2tyuiopf',path:'base2.html'},
	{id:7,title:'test2',content:'内容2tyuiopf',path:'base2.html'},
	{id:8,title:'test2',content:'内容2tyuiopf',path:'base2.html'},
	{id:9,title:'test9',content:'内容qwertyui4',path:'base5.html'}
];


function getCfg(){
	var columns = [
		{field:'id',title:'ID',width:30,fitColumn:false},
		{field:'title',title:'标题',width:60},
		{field:'content',title:'内容',width:100,textLimit:true},
		{field:'path',title:'文件名',width:60,textLimit:true}
	];
	var options = {
		title : '行锁',
		data : data,
		lockRows : [2,3],//锁定 3，4行
		rowNumbersWidth : '25px',
		textLimit:true,//开启
		columns : columns
	}
	return options;
}
var grid1;
function create(){
	var opt1 = getCfg();
	grid1 = $("#demo1").extgrid(opt1);
}
$(function(){
	create();
});
