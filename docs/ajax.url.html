function getCfg(){
	var columns = [
		{field:'id',title:'ID',width:30,sortable:true},
		{field:'title',title:'标题',width:60},
		{field:'content',title:'内容',width:200},
		{field:'path',title:'文件名',width:100}
	];
	var options = {
		title : 'url获取',
		url : 'datagrid_data.json',
		columns : columns,
		pagination : true
	}
	return options;
}
var grid1;
function create(){
	var opt1 = getCfg();
	grid1 = $("#demo1").extgrid(opt1);
}
$(function(){
	create();
});
