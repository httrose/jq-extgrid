Grid 行样式:rowStyler

默认为"" | 字符,函数

CSS 
.color { background: #DACCFD}
.color1 { background:#FFFFCC}
.color2 { background: #FF00FF}
HTML:
Script: 
var data = [
	{id:1,title:'test1',content:'内容1234566789',path:'base1.html'},
	{id:2,title:'test2',content:'内容2tyuiopf',path:'base2.html'},
	{id:3,title:'test3',content:'内容3zxcvbnm',path:'base3.html'},
	{id:4,title:'test4',content:'内容qwertyui4',path:'base4.html'}
];


function getCfg(){
	var columns = [
		{field:'id',title:'ID',width:30},
		{field:'title',title:'标题',width:60},
		{field:'content',title:'内容',width:100,textLimit:true},
		{field:'path',title:'文件名',width:60,textLimit:true}
	];
	var options = {
		title : '行样式',
		data : data,
		rowStyler : "color",//开启单选
		textLimit:true,//开启
		columns : columns
	}
	return options;
}
var grid1,grid2;
function create(){
	var opt1 = getCfg();
	grid1 = $("#demo1").extgrid(opt1);
	var opt2 = getCfg();
	opt2.rowStyler = function(tr,rowId,rowData){
		var grid = this;
		if(rowId%2) {
			$(tr).addClass("color1");
		} else {
			$(tr).addClass("color2");
		}
	};
	grid2 = $("#demo2").extgrid(opt2);
}
$(function(){
	create();
});
