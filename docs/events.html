onUnselectAll : self.fireEvent('onUnselectAll',self,[]);//返回false不取消选择
onHeaderContextMenu :self.fireEvent('onHeaderContextMenu',self,[this,e]);//当 datagrid 的头部被右键单击时触发
onHeaderCreate : self.fireEvent('onHeaderCreate',self,[]);//当 grid-header 创建完成时调用
onToolBarCreate: self.fireEvent('onToolBarCreate',self,[self.configs.views['toolbar'],opt.toolBars]);//工具栏创建后调用
onRowContextMenu : self.fireEvent('onRowContextMenu',self,[this,rowId,rowData,e]);//this:当前tr  当右键点击行时触发

onRefresh : self.fireEvent('onRefresh',self);
onChangePageSize : self.fireEvent('onChangePageSize',self,[pageSize]); //返回false不执行

onBeforeShowGrid :self.fireEvent('onBeforeShowGrid',self,[]);
onGetData :self.fireEvent('onGetData',self,[opt.data]);// grid 数据变动都会调用
onPagerCreate : self.fireEvent("onPagerCreate",self,[opt.views['pager']]);

onClickRowNumber : self.fireEvent('onClickRowNumber',self,[rid,e]);

onUnlockColumn : self.fireEvent('onUnlockColumn',self,[field]);//返回false取消执行
onUnlockRow : self.fireEvent('onUnlockRow',self,[field]);//返回false取消执行
onViewSizeChange : self.fireEvent("onViewSizeChange",self,[]);//视图大小改变
onSizeChange : self.fireEvent("onSizeChange",self,[width,height]);//改变gird大小
onScroll : self.fireEvent("onScroll",self,[]);
onDataChange :self.fireEvent("onDataChange",self,[datas]);//数据有变更
onCellEdit : self.fireEvent('onCellEdit',self,[t,rid,field,rowData]);//t:div.cell 单元格数据有变更调用

onUpdate : $.noop,//更新数据
onDelete : $.noop,//删除数据
