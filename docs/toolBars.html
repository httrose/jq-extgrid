var data = [
	{id:1,title:'test1',content:'内容1',path:'base1.html'},
	{id:2,title:'test2',content:'内容2',path:'base2.html'},
	{id:3,title:'test3',content:'内容3',path:'base3.html'},
	{id:4,title:'test4',content:'内容4',path:'base4.html'}
];


function getCfg(){
	var tools = [
		{text:'添加',cls:'icon-add',callBack:function(){var grid = this; alert("添加");}},
		{text:'修改',cls:'icon-edit',callBack:function(){var grid = this; alert("修改");}},
		"|",//可以是 "," "-"
		{text:'删除',cls:'icon-del',callBack:function(t){
			var grid = this;
			if( $(t).hasClass("l-btn-disabled") ) {
				alert("当前按钮已经被禁止");
			}
		},disabled:true},
		"|",
		"
当前是按钮栏
"
	];
	var columns = [
		{field:'id',title:'ID',width:30,fitColumn:false},
		{field:'title',title:'标题',width:60},
		{field:'content',title:'内容',width:200},
		{field:'path',title:'文件名',width:100}
	];
	var options = {
		title : 'gird工具栏',
		data : data,
		toolBars :　tools,
		columns : columns
	}
	return options;
}
var grid1;
function create(){
	var opt1 = getCfg();
	grid1 = $("#demo1").extgrid(opt1);
}
$(function(){
	create();
});
