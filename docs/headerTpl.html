];
var htpl = $("#tpl").html();

function getCfg(){
	var columns = [
		{field:'id',title:'ID',width:30},
		{field:'title',title:'标题',width:60},
		{field:'content',title:'内容',width:100,textLimit:true},
		{field:'path',title:'文件名',width:60,textLimit:true}
	];
	var options = {
		title : '自定义列模板实现列合并',
		data : data,
		headerTpl : htpl,
		textLimit:true,//开启
		columns : columns
	}
	return options;
}
var grid1,grid2;
function create(){
	var opt1 = getCfg();
	grid1 = $("#demo1").extgrid(opt1);
}
$(function(){
	create();
});
